5 最长回文子串

字符串

https://leetcode-cn.com/problems/longest-palindromic-substring/

给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。

示例 1：

1
2
3

输入: "babad"
输出: "bab"
注意: "aba" 也是一个有效答案。

示例 2：

1
2

输入: "cbbd"
输出: "bb"

回文串

“回文串”是一个正读和反读都一样的字符串，比如“level”或者“noon”等等就是回文串。
最长回文子串即是一个字符串中最长的一个回文子字符串。

中心扩散算法

算法思想

该算法的思维是遍历字符串，在当前字符向两端扩散，如果符合回文字串的定义则记录下回文子串的两端 index，遍历过程中不断比较子串长度，遍历结束即获得最长回文子串。

public String longestPalindrome(String s){
    if (s.length() < 2){
        return s;
    }
    int[] range = new int[2];
    char[] cArray = s.toCharArray();
    for (int i = 0; i < cArray.length; i++){ // 根据回文串定义进行遍历
        i = findNext(cArray, i, range);
    }
    return s.substring(range[0], range[1]+1);
}

public int findNext(char[] cArray, int low, int[] range) {
    int high = low;
    while (high + 1 < cArray.length && cArray[low] == cArray[high+1]){ // 根据相同字符 向右合并
        high++;
    }
    int ans = high;
    while (low > 0 && high + 1 < cArray.length && cArray[low-1] == cArray[high+1]){  // 扩散 
        low--;
        high++;
    }

    if (high-low > range[1]-range[0]){  // 判断长度
        range[1] = high;
        range[0] = low;
    }
    
    return ans;
}

public String longestPalindrome2(String s){
    int len = s.length();
    if (len < 2){
        return s;
    }
    // 初始化
    int maxLen = 1;
    int begin = 0;
    boolean[][] dp = new boolean[len][len];  // 定义状态
    for (int i = 0; i < len; i++){
        dp[i][i] = true;
    }
    char[] cArray = s.toCharArray();
    for (int j = 0; j < len; j++){
        for (int i = 0; i < j; i++){
            if (cArray[i] == cArray[j]){
                if (j - i < 3){ // 0 1 2 子串是最小回文串
                    dp[i][j] = true;
                }else {
                    dp[i][j] = dp[i+1][j-1];  // 状态转移
                }
            }else {
                dp[i][j] = false;
            }

            if (dp[i][j] && j-i+1 > maxLen) {  // 判断长度
                begin = i;
                maxLen = j-i+1;
            }
        }
    }
    return s.substring(begin, begin+maxLen);


}
